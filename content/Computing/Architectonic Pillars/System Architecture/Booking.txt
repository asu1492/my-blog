@Entity 
@Table
public class Show{
    @Id
    @GeneratedValue(stategy=GenerationType.IDENTITY)
    private Long id;

    private String movieNamel;
    private LocalDateTime showTime;
}

@Entity
public class Seat{
    
    @Id
    @GeneratedValue(stategy=GenerationType.IDENTITY)
    private Long id;

    private String seatNumber;

    @ManyToOne
    private Show show;
}
